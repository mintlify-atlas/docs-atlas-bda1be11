Tasks are the individual work items in your TaskFlow boards. Each task belongs to a column and can be moved between columns to track progress.

## Understanding Tasks

A task represents a single work item or action item. Each task has:

- **Title**: A brief description of the task
- **Description**: Detailed information about the task
- **Column**: The column where the task currently resides
- **Position**: The task's order within its column

### Task Data Structure

```typescript
export interface TaskItem {
  id: number;
  title: string;
  description: string;
}
```

## Creating a Task

<Steps>
  <Step title="Set Up Task Service">
    Inject the `TaskService` into your column component:

    ```typescript
    import { TaskService } from '../../../services/task.service';
    import { TaskItem } from '../../../services/board.service';

    export class Column {
      private taskService = inject(TaskService);
      
      showModal = false;
      selectedTask: TaskItem | null = null;
      modalMode: 'create' | 'edit' | 'view' = 'create';
    }
    ```
  </Step>

  <Step title="Open Task Creation Modal">
    ```typescript
    openCreateModal() {
      this.selectedTask = null;
      this.modalMode = 'create';
      this.showModal = true;
    }

    closeModal() {
      this.showModal = false;
    }
    ```
  </Step>

  <Step title="Handle Task Creation">
    Create the task and emit an event to update the parent board:

    ```typescript
    handleSave(data: { title: string; description: string }) {
      if (this.modalMode === 'create') {
        this.taskService.createTask({
          columnId: this.column.id,
          title: data.title,
          description: data.description
        }).subscribe({
          next: task => {
            // Emit event to parent component
            this.taskCreated.emit({
              columnId: this.column.id,
              task
            });

            this.closeModal();
          },
          error: err => console.error('Error creating task:', err)
        });
      }
    }
    ```
  </Step>

  <Step title="Update Parent Board State">
    Handle the task creation event in the parent board component:

    ```typescript
    // In BoardDetailPage component
    handleTaskCreated(event: { columnId: number; task: TaskItem }) {
      this.board.update(b => {
        if (!b) return b;

        return {
          ...b,
          columns: b.columns.map(c =>
            c.id === event.columnId
              ? { ...c, tasks: [...c.tasks, event.task] }
              : c
          )
        };
      });
    }
    ```
  </Step>

  <Step title="Add Task Modal Template">
    ```html
    <button class="add-task-btn" (click)="openCreateModal()">
      + New Task
    </button>

    @if (showModal) {
      <app-task-modal
        [task]="selectedTask"
        [mode]="modalMode"
        (close)="closeModal()"
        (save)="handleSave($event)">
      </app-task-modal>
    }
    ```
  </Step>
</Steps>

## Task Modal Component

The task modal handles both creation and editing:

<CodeGroup>
```typescript task-modal.ts
import { Component, Input, Output, EventEmitter, signal } from '@angular/core';
import { TaskItem } from '../../../services/board.service';

@Component({
  selector: 'app-task-modal',
  templateUrl: './task-modal.html',
  styleUrl: './task-modal.css',
})
export class TaskModal {
  @Input() task: TaskItem | null = null;
  @Input() mode: 'create' | 'edit' | 'view' = 'create';

  @Output() close = new EventEmitter<void>();
  @Output() save = new EventEmitter<{
    title: string;
    description: string;
  }>();

  title = signal('');
  description = signal('');
  isSaving = false;

  ngOnInit() {
    if (this.task) {
      this.title.set(this.task.title);
      this.description.set(this.task.description || '');
    }
  }

  onSave() {
    if (!this.title().trim()) return;

    this.isSaving = true;

    this.save.emit({
      title: this.title(),
      description: this.description()
    });
  }

  onClose() {
    this.close.emit();
  }
}
```

```html task-modal.html
<div class="modal-overlay" (click)="onClose()">
  <div class="modal-content" (click)="$event.stopPropagation()">
    <h2>{{ mode === 'create' ? 'New Task' : 'Edit Task' }}</h2>

    <input
      type="text"
      placeholder="Task title..."
      [value]="title()"
      (input)="title.set($event.target.value)"
      autofocus
    />

    <textarea
      placeholder="Task description..."
      [value]="description()"
      (input)="description.set($event.target.value)"
      rows="5"
    ></textarea>

    <div class="modal-actions">
      <button (click)="onSave()" [disabled]="isSaving || !title().trim()">
        {{ mode === 'create' ? 'Create' : 'Save' }}
      </button>
      <button (click)="onClose()" [disabled]="isSaving">
        Cancel
      </button>
    </div>
  </div>
</div>
```
</CodeGroup>

## Displaying Tasks

Render tasks within a column using the task card component:

```html
<div class="tasks">
  @for (task of column.tasks; track task.id) {
    <app-task-card
      [task]="task"
      (click)="openEditModal(task)"
      (delete)="deleteTask(task.id)">
    </app-task-card>
  }

  @if (column.tasks.length === 0) {
    <p class="empty-state">No tasks</p>
  }
</div>
```

### Task Card Component

<CodeGroup>

## Editing a Task

<Steps>
  <Step title="Open Edit Modal">
    ```typescript
    openEditModal(task: TaskItem) {
      this.selectedTask = task;
      this.modalMode = 'edit';
      this.showModal = true;
    }
    ```
  </Step>

  <Step title="Handle Task Update">
    ```typescript
    handleSave(data: { title: string; description: string }) {
      // ... create logic ...

      if (this.modalMode === 'edit' && this.selectedTask) {
        this.taskService.updateTask(this.selectedTask.id, {
          title: data.title,
          description: data.description
        }).subscribe({
          next: updated => {
            this.taskUpdated.emit({
              columnId: this.column.id,
              task: updated
            });

            this.closeModal();
          },
          error: err => console.error('Error updating task:', err)
        });
      }
    }
    ```
  </Step>

  <Step title="Update Board State">
    ```typescript
    // In BoardDetailPage component
    handleTaskUpdated(event: { columnId: number; task: TaskItem }) {
      this.board.update(b => {
        if (!b) return b;

        return {
          ...b,
          columns: b.columns.map(c =>
            c.id === event.columnId
              ? {
                  ...c,
                  tasks: c.tasks.map(t =>
                    t.id === event.task.id ? event.task : t
                  )
                }
              : c
          )
        };
      });
    }
    ```
  </Step>
</Steps>

## Deleting a Task

<Warning>
  Task deletion is permanent and cannot be undone.
</Warning>

<Steps>
  <Step title="Delete from Column Component">
    ```typescript
    deleteTask(taskId: number) {
      this.taskService.deleteTask(taskId).subscribe(() => {
        this.taskDeleted.emit({
          columnId: this.column.id,
          taskId
        });
      });
    }
    ```
  </Step>

  <Step title="Handle Deletion in Parent">
    ```typescript
    // In BoardDetailPage component
    handleTaskDeleted(event: { columnId: number; taskId: number }) {
      this.board.update(b => {
        if (!b) return b;

        return {
          ...b,
          columns: b.columns.map(c =>
            c.id === event.columnId
              ? {
                  ...c,
                  tasks: c.tasks.filter(t => t.id !== event.taskId)
                }
              : c
          )
        };
      });
    }
    ```
  </Step>

  <Step title="Add Delete Button">
    ```html
    <app-task-card
      [task]="task"
      (click)="openEditModal(task)"
      (delete)="deleteTask(task.id)">
    </app-task-card>
    ```
  </Step>
</Steps>

## Moving Tasks Between Columns

Tasks can be moved between columns using the `moveTask` API:

```typescript
export class TaskService {
  moveTask(id: number, data: { newColumnId: number; newPosition: number }) {
    return this.http.put(`${this.apiUrl}/${id}/move`, data);
  }
}
```

### Manual Move Example

```typescript
moveTaskToColumn(taskId: number, targetColumnId: number, position: number) {
  this.taskService.moveTask(taskId, {
    newColumnId: targetColumnId,
    newPosition: position
  }).subscribe(() => {
    // Refresh board or update local state
    this.loadBoard();
  });
}
```

<Note>
  For drag-and-drop functionality, see the [Drag and Drop Guide](/guides/drag-and-drop).
</Note>

## Complete Task Service API

<CodeGroup>
```typescript task.service.ts
import { Injectable, inject } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { TaskItem } from './board.service';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class TaskService {
  private http = inject(HttpClient);
  private apiUrl = 'https://localhost:7057/api/task';

  // Create a new task
  createTask(data: {
    title: string;
    columnId: number;
    description?: string;
  }) {
    return this.http.post<TaskItem>(this.apiUrl, data);
  }

  // Update an existing task
  updateTask(id: number, data: { title: string; description: string }) {
    return this.http.put<TaskItem>(`${this.apiUrl}/${id}`, data);
  }

  // Delete a task
  deleteTask(id: number) {
    return this.http.delete(`${this.apiUrl}/${id}`);
  }

  // Move task to different column/position
  moveTask(id: number, data: { newColumnId: number; newPosition: number }) {
    return this.http.put(`${this.apiUrl}/${id}/move`, data);
  }
}
```
</CodeGroup>

## Event Communication Pattern

TaskFlow uses a hierarchical event system:

<Tabs>
  <Tab title="Task Card">
    Emits basic events:

    ```typescript
    @Output() delete = new EventEmitter<number>();

    onDelete(event: Event) {
      event.stopPropagation();
      this.delete.emit();
    }
    ```
  </Tab>

  <Tab title="Column Component">
    Handles task events and re-emits with context:

    ```typescript
    @Output() taskDeleted = new EventEmitter<{
      columnId: number;
      taskId: number;
    }>();

    deleteTask(taskId: number) {
      this.taskService.deleteTask(taskId).subscribe(() => {
        this.taskDeleted.emit({
          columnId: this.column.id,
          taskId
        });
      });
    }
    ```
  </Tab>

  <Tab title="Board Component">
    Updates the board state:

    ```typescript
    handleTaskDeleted(event: { columnId: number; taskId: number }) {
      this.board.update(b => {
        if (!b) return b;
        return {
          ...b,
          columns: b.columns.map(c =>
            c.id === event.columnId
              ? { ...c, tasks: c.tasks.filter(t => t.id !== event.taskId) }
              : c
          )
        };
      });
    }
    ```
  </Tab>
</Tabs>

## Best Practices

<AccordionGroup>
  <Accordion title="Use Signals for Form State">
    Use Angular signals for reactive form state in modals:

    ```typescript
    title = signal('');
    description = signal('');

    onSave() {
      this.save.emit({
        title: this.title(),
        description: this.description()
      });
    }
    ```
  </Accordion>

  <Accordion title="Prevent Event Bubbling">
    Stop event propagation to prevent unintended actions:

    ```typescript
    onDelete(event: Event) {
      event.stopPropagation();
      this.delete.emit();
    }
    ```
  </Accordion>

  <Accordion title="Validate Before Save">
    Always validate task data before submitting:

    ```typescript
    onSave() {
      if (!this.title().trim()) {
        // Show error or prevent save
        return;
      }

      this.save.emit({
        title: this.title(),
        description: this.description()
      });
    }
    ```
  </Accordion>

  <Accordion title="Handle Loading States">
    Disable buttons during save operations:

    ```html
    <button 
      (click)="onSave()" 
      [disabled]="isSaving || !title().trim()">
      {{ isSaving ? 'Saving...' : 'Save' }}
    </button>
    ```
  </Accordion>

  <Accordion title="Use Track By for Performance">
    Always use `track` in `@for` loops for better performance:

    ```html
    @for (task of column.tasks; track task.id) {
      <app-task-card [task]="task"></app-task-card>
    }
    ```
  </Accordion>
</AccordionGroup>
